import pandas as pd

def topLearnerCourseTransitions(course_completions: pd.DataFrame) -> pd.DataFrame:
    black_df = course_completions.copy()
    black_df['completion_date'] = pd.to_datetime(black_df['completion_date'])
    
    black_stats = black_df.groupby('user_id').agg(
        black_cnt=('course_id', 'count'),
        black_avg=('course_rating', 'mean')
    ).reset_index()
    
    black_top_ids = black_stats[
        (black_stats['black_cnt'] >= 5) & (black_stats['black_avg'] >= 4)
    ]['user_id']
    
    black_filtered = black_df[black_df['user_id'].isin(black_top_ids)].copy()
    black_filtered = black_filtered.sort_values(['user_id', 'completion_date'])
    
    black_filtered['first_course'] = black_filtered['course_name']
    black_filtered['second_course'] = black_filtered.groupby('user_id')['course_name'].shift(-1)
    
    black_pairs = black_filtered.dropna(subset=['second_course'])
    
    black_result = black_pairs.groupby(['first_course', 'second_course']).size().reset_index(name='transition_count')
    
    black_result['lower_first'] = black_result['first_course'].str.lower()
    black_result['lower_second'] = black_result['second_course'].str.lower()
    
    black_result = black_result.sort_values(
        ['transition_count', 'lower_first', 'lower_second'],
        ascending=[False, True, True]
    )
    
    return black_result[['first_course', 'second_course', 'transition_count']]
