import pandas as pd

def tree_node(tree: pd.DataFrame) -> pd.DataFrame:
    black = set(tree['p_id'].dropna())
    def classify(row):
        if pd.isna(row['p_id']): return 'Root'
        if row['id'] in black: return 'Inner'
        return 'Leaf'
    tree['type'] = tree.apply(classify, axis=1)
    return tree[['id', 'type']]
