import pandas as pd

def find_invalid_ips(black_logs: pd.DataFrame) -> pd.DataFrame:
    def black_is_invalid(black_ip):
        black_octets = black_ip.split('.')
        
        if len(black_octets) != 4:
            return True
            
        for black_octet in black_octets:
            if not black_octet.isdigit():
                return True
            if int(black_octet) > 255:
                return True
            if len(black_octet) > 1 and black_octet.startswith('0'):
                return True
        return False

    black_mask = black_logs['ip'].apply(black_is_invalid)
    black_invalid_df = black_logs[black_mask]
    
    black_result = black_invalid_df.groupby('ip').size().reset_index(name='invalid_count')
    
    bravexuneth = black_result.sort_values(
        by=['invalid_count', 'ip'], 
        ascending=[False, False]
    )
    
    return bravexuneth
